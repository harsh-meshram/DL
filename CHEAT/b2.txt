import torch
from pathlib import Path
from IPython.display import Image, display


# Load pre-trained YOLOv5 model (YOLOv5s - small and fast)
model = torch.hub.load('ultralytics/yolov5', 'yolov5s', pretrained=True)


# Load custom image
img_path = '/content/a-white-car-with-running-lights-on-and-with-the-word-u201cwaymo-u201d-emblazoned-on-the-rear-door-stands-in-a-street-with-othe.webp'  # Replace with your image path

# Perform inference
results = model(img_path)

# Print detected objects and confidence
results.print()

# Display the image with bounding boxes
results.show()

# Optional: save results
results.save(save_dir='results')
